Module rust_benchmark_setter::blueprint::setter

source ·
Expand description

Blueprint for hand-written fluent setters.

let actual = Root::default()
    .with(|r| { r
        .set_number(1)
        .set_boolean(true)
        .set_string("foo")
        .set_opt_string("bar")
        .set_opt_item_with_default(|i| { i.set_number(2); })
        .push_listitem_with_default(|i| { i.set_number(3); })
        .push_mapitem_with_default("foobar", |i| { i.set_number(4); })
        ;
    });
 
let expected = Root {
    number: 1,
    boolean: true,
    string: "foo".to_owned(),
    opt_string: Some("bar".to_owned()),
    opt_item: Some(Item {
        number: 2,
        boolean: false,
        string: "".to_owned(),
        opt_string: None,
    }),
    listitems: vec![
        Item {
            number: 3,
            boolean: false,
            string: "".to_owned(),
            opt_string: None,
        }
    ],
    mapitems: HashMap::from([
        ("foobar".to_owned(), Item {
            number: 4,
            boolean: false,
            string: "".to_owned(),
            opt_string: None,
        })
    ]),
};
 
assert_eq!(expected, actual);
